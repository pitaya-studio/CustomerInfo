首页跳转至产品列表页（原首页）

描述:

接口路径:/t1/jumpParam

输入(param):

                {
                "jumpType":1,//跳转类型(当类型为2和3的时候只返回选择的供应商)
                "travelSelected":{travelArea:[],//区域id（东南亚等）
                                  travelInOut:[],//国内和出境id
                                  travelId:[]//供应商id}//选择的供应商或者区域，没有选择传空
                                  },
                "tabText":""//搜索内容
                }

输出(data)

          {
            "results":{
                travelSelected:[
                    {
                        travelAreaId:"",
                        travelAreaName:""//当传入的id为空时返回空
                     },
                     {
                        travelInOutId:"",
                        travelInOutName:""//当传入的id为空时返回空
                     },
                     {
                        travelId:"",
                        travelName:""//当传入的id为空时返回空
                     }
                ],
                jumpType:""，
                "tabText":""//搜索内容
             }
        }
